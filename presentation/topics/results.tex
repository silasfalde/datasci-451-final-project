\section{Results}

\begin{frame}{Expected Findings and Interpretation Plan}
	\begin{itemize}
		\item We expect stronger vaccination probability when respondents report:
		\begin{itemize}
			\item doctor recommendation,
			\item higher perceived risk,
			\item higher perceived vaccine effectiveness,
			\item healthcare worker status.
		\end{itemize}
		\item We expect behavioral prevention variables to provide additional predictive signal.
		\item Interpretation emphasis:
		\begin{itemize}
			\item quantify directional effects and uncertainty using credible intervals,
			\item compare consistency of patterns across H1N1 and seasonal outcomes,
			\item identify subgroups where uncertainty remains large.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Limitations, Ethics, and Next Steps}
	\begin{itemize}
		\item Known data limitations:
		\begin{itemize}
			\item self-report bias and recall error,
			\item non-response bias from phone-survey collection,
			\item no direct medical verification in this dataset.
		\end{itemize}
		\item Ethical interpretation plan:
		\begin{itemize}
			\item avoid causal claims from observational survey data,
			\item examine subgroup uncertainty before policy recommendations.
		\end{itemize}
		\item Next project milestones: finalize preprocessing, fit planned models, complete posterior checks, and prepare validated comparison results.
	\end{itemize}
\end{frame}
